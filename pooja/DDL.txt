USE ftp109;
CREATE TABLE CUSTOMER(
     CUS_ID INT UNSIGNED PRIMARY KEY,
     CUS_NAME VARCHAR(20) NOT NULL,
     CONTACT_NO INT,
     PASSWORD VARCHAR(20) NOT NULL,
     CUS_WALLET INT DEFAULT 0
	);


CREATE TABLE VENDOR(
	VEN_ID INT UNSIGNED PRIMARY KEY,
     VEN_NAME VARCHAR(20) NOT NULL,
     CONTACT_NO INT,
     PASSWORD VARCHAR(20) NOT NULL,
     VEN_WALLET INT DEFAULT 0
	);
    
    
    
CREATE TABLE MENU(
	FOD_ID INT UNSIGNED PRIMARY KEY,
     FOD_NAME VARCHAR(20) NOT NULL,
	FOD_TYPE VARCHAR(20) NOT NULL,
    FOD_PRICE INT NOT NULL
	);

CREATE TABLE ORDR(
       ORD_ID INT UNSIGNED PRIMARY KEY,
       ORD_STATUS VARCHAR(10) NOT NULL,
       ORD_PRICE INT NOT NULL,
       ORD_DAT DATETIME, 
	   CUS_IDO INT UNSIGNED,
	   VEN_IDO INT UNSIGNED,
       CONSTRAINT CUS_ID_FK FOREIGN KEY (CUS_IDO) REFERENCES CUSTOMER(CUS_ID),
       CONSTRAINT VEN_ID_FK FOREIGN KEY (VEN_IDO) REFERENCES VENDOR(VEN_ID)
       );

       
CREATE TABLE MEN_ORDR(
       FOD_IDF INT UNSIGNED NOT NULL, 
       ORD_IDF INT UNSIGNED NOT NULL,
	   CONSTRAINT FOD_IDF_FK FOREIGN KEY (FOD_IDF) REFERENCES MENU(FOD_ID),
       CONSTRAINT ORD_IDF_FK FOREIGN KEY (ORD_IDF) REFERENCES ORDR(ORD_ID)
       );
